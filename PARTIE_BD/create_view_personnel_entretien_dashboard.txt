CREATE OR REPLACE VIEW V_PERSONNEL_ENTRETIEN_DASHBOARD AS
SELECT
    p.ID_PERSONNEL,
    p.NOM,
    p.PRENOM,
    f.NOM_FONCTION,

    sup.NOM || ' ' || sup.PRENOM AS NOM_SUPERIEUR,
    rem.NOM || ' ' || rem.PRENOM AS NOM_REMPLACANT,

    COUNT(DISTINCT e.ID_ENCLOS) AS NB_ENCLOS_COUVERTS,

    COUNT(DISTINCT r.ID_REPARATION) AS NB_REPARATIONS_TOTAL,

    MAX(r.DATE_DEBUT_REPARATION) AS DERNIERE_INTERVENTION,

FROM PERSONNEL p

JOIN FONCTION f ON f.ID_FONCTION = p.ID_FONCTION

LEFT JOIN PERSONNEL sup ON sup.ID_PERSONNEL = p.ID_SUPERIEUR
LEFT JOIN PERSONNEL rem ON rem.ID_PERSONNEL = p.ID_REMPLACANT

LEFT JOIN EST_AFFECTEE_A ea ON ea.ID_PERSONNEL = p.ID_PERSONNEL
LEFT JOIN ZONE z ON z.ID_ZONE = ea.ID_ZONE
LEFT JOIN ENCLOS e ON e.ID_ZONE = z.ID_ZONE

LEFT JOIN REPARATION r ON r.ID_PERSONNEL = p.ID_PERSONNEL

WHERE p.ID_FONCTION = 16
  AND p.ESTARCHIVE = 1

GROUP BY
    p.ID_PERSONNEL,
    p.NOM,
    p.PRENOM,
    f.NOM_FONCTION,
    sup.NOM,
    sup.PRENOM,
    rem.NOM,
    rem.PRENOM;
